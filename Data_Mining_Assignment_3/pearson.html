Pearson Correlation Coefficient Pseudocode 


// Between samples x, y
//
//   x = x[0] .. x[N-1], y = y[0] .. y[N-1]
//
// use online variance calculation to compute Pearson's r in a
// single pass.
//
// Cov(X,Y) = E[(X-E[X])(Y-E[Y])] = E[XY]-E[X]E[Y]
// Sx = sqrt(E[X^2]-E[X]^2)

function pearson(x,y){
    sum_sq_x = 0
    sum_sq_y = 0
    sum_coproduct = 0
    mean_x = 0
    mean_y = 0
    for (i = 1 to N) {
        sum_sq_x += x[i] * x[i]
        sum_sq_y += y[i] * y[i]
        sum_coproduct += x[i] * y[i]
        mean_x += x[i]
        mean_y += y[i]
    }
    mean_x = mean_x / N
    mean_y = mean_y / N
    pop_sd_x = sqrt((sum_sq_x/N) - (mean_x * mean_x))
    pop_sd_y = sqrt((sum_sq_y / N) - (mean_y * mean_y))
    cov_x_y = (sum_coproduct / N) - (mean_x * mean_y)
    correlation = cov_x_y / (pop_sd_x * pop_sd_y)
    return correlation
}
